Usage

forms.py

       
from django import forms
from parsley.decorators import parsleyfy

@parsleyfy
class FieldTypeForm(forms.Form):
  name = forms.CharField(min_length=3, max_length=30)
  url = forms.URLField()
  url2 = forms.URLField(required=False)
  email = forms.EmailField()
  email2 = forms.EmailField(required=False)
  age = forms.IntegerField()
  income = forms.DecimalField()

views.py

       
from django.views.generic.edit import FormView
from core.forms import FieldTypeForm

class HomeView(FormView):
  template_name = "simpleform.html"
  form_class = FieldTypeForm
  success_url = '/'
